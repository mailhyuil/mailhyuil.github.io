victor

js 벡터 구현 라이브러리

install​

yarn add victor
yarn add @types/victor

es6​

import Victor from "victor";

usage​

const pos = Victor(0,0)
const vel = Victor(1,1)
const acc = Victor(0,0)

pos.add(vel)
vel.add(acc)
