# c++ type

```cpp
int i = 4;              // 정수형 변수. short 없음
float f = 4.0;          // 실수형 변수
double d = 4.0;         // 실수형 변수
bool b = true;          // 불린형 변수

char c = 'c';           // 문자형 변수
string hello = "Hello"; // string 변수 char[] 대신 사용

int arr[5];             // 정수형 배열
int arr[] = {1, 2, 3};  // 정수형 배열

int* arr[100];          // 정수형 포인터 배열
arr[0] = 1;             // 정수형 포인터 배열에 값 할당

int* arr = new int[100]; // 동적할당
```
