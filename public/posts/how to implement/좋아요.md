## prisma

```prisma
model users {
    id        String    @id @default(uuid())
    username  String    @unique
    password  String
    likes     likes[]
    posts  posts[]
}

model posts {
    id        String    @id @default(uuid())
    title     String
    content   String
    user      users     @relation(fields: [userId], references: [id])
    userId    String
    likes     likes[]
    likes_count likes_count
}

model likes {
    id        String    @id @default(uuid())
    user      users     @relation(fields: [userId], references: [id])
    userId    String
    post   posts  @relation(fields: [postId], references: [id])
    postId String
}

model likes_count {
    id        String    @id @default(uuid())
    post   posts  @relation(fields: [postId], references: [id])
    postId String
    count Int
}
```
